## Usage

```tsx
import {
  TreeView,
  TreeViewBranch,
  TreeViewLeaf,
} from "@/components/ui/tree-view";

export default function MyComponent() {
  return (
    <TreeView defaultExpandedItems={["src"]}>
      <TreeViewBranch id="src" name="src">
        <TreeViewLeaf id="app.tsx" name="app.tsx" />
        <TreeViewLeaf id="main.tsx" name="main.tsx" />
      </TreeViewBranch>
      <TreeViewLeaf id="package.json" name="package.json" />
    </TreeView>
  );
}
```
